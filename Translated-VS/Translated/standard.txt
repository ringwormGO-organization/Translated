#include <iostream>
#include "Class-example.hpp"

Square square;
Rectangle rectangle;

void Square_Extent()
{
	std::cout << "Type a length of side: " << std::endl;
	std::cin >> square.side;

	std::cout << "Extent of square is: " << square.side * 4 << std::endl;
}

void Rectangle_Extent()
{
	std::cout << "Type a length of first side: " << std::endl;
	std::cin >> rectangle.side1;

	std::cout << "Type a length of second side: " << std::endl;
	std::cin >> rectangle.side2;

	rectangle.Extent = rectangle.side1 * 2 + rectangle.side2 * 2;

	std::cout << "Extent of square is: " << rectangle.Extent << std::endl;
}
